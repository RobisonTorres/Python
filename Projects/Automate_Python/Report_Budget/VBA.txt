First Module

Record the macro and apply "zoom selection" on Dash (='Gráfico - Anual'!$A$1:$P$26)

Sub AdjustZoom()
    On Error GoTo Sair
    
    ActiveSheet.Range("Dash").Select
    ActiveWindow.Zoom = True

Sair:
    Exit Sub
End Sub

Second Module

Sub AutoSelectCurrentMonth()
    Dim ws As Worksheet
    Dim cell As Range
    Dim currentMonth As String
    
    ' Set the worksheet and cell where you have data validation
    Set ws = ThisWorkbook.Sheets("Gráfico - Mensal") ' Change "Sheet1" to your sheet name
    Set cell = ws.Range("B3") ' Change "A1" to your cell reference
    
    ' Get the current month with the first letter capitalized
    currentMonth = WorksheetFunction.Proper(Format(Date, "mmmm"))
    cell.Value = currentMonth
    
End Sub

On thisworkfolder

Private Sub Workbook_SheetActivate(ByVal Sh As Object)
    AdjustZoom
    AutoSelectCurrentMonth
End Sub
